🛠 Step 3: Build a Simple Generator (Python)
from openai import OpenAI
import json

client = OpenAI(api_key="YOUR_KEY")

def load_knowledge():
    with open("knowledge-pack/error_handling.txt") as f:
        error = f.read()
    with open("knowledge-pack/logging_patterns.txt") as f:
        logging = f.read()
    with open("knowledge-pack/subflow_catalog.json") as f:
        subflows = json.load(f)
    return error, logging, subflows

def generate_flow(user_input):
    error, logging, subflows = load_knowledge()

    prompt = f"""
    You are an IBM ACE expert.

    Error handling:
    {error}

    Logging:
    {logging}

    Subflows:
    {subflows}

    Requirement:
    {user_input}

    Generate msgflow XML and ESQL.
    """

    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}]
    )

    return response.choices[0].message.content

print(generate_flow("Consume REST API and write to MQ"))
